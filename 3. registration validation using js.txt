    <script>
      function validate() {
        var name = document.forms["form"]["name"];
        var age = document.forms["form"]["age"];
        var email = document.forms["form"]["email"];
        var password = document.forms["form"]["psw"];
        var gender = document.forms["form"]["gender"];
        if (name.value == "" || name.value == null) {
          alert("Name should not be empty");
          name.focus();
          return false;
        }
        if (age.value == "" || age.value == null) {
          alert("Age should not be empty");
          age.focus();
          return false;
        }
        if (isNaN(age.value)) {
          alert("Age should in Number");
          age.focus();
          return false;
        }
        if (email.value == "" || email.value == null) {
          alert("Email should not be empty");
          email.focus();
          return false;
        }
        if (email.value.indexOf("@") == -1 || email.value.indexOf(".") == -1 || email.value.indexOf(".") == 0) {
          alert("Invalid Email ID");
          email.focus();
          return false;
        }
        if (password.value == "") {
          alert("Password should not be empty");
          password.focus();
          return false;
        }
        if (password.value.length < 5) {
          alert("Length of the password must be greater than 4");
          password.focus();
          return false;
        }
        if (gender[0].checked == false && gender[1].checked == false) {
          alert("Select Gender");
          return false;
        }
        alert("Your Details are correct");
        return true;
      }
